Schema — CREATE TABLE
-- Books table CREATE TABLE books ( book_id INT PRIMARY KEY AUTO_INCREMENT, title VARCHAR(255) NOT NULL, author VARCHAR(255) NOT NULL, year YEAR, genre VARCHAR(100), status ENUM('available', 'borrowed') DEFAULT 'available' ); -- Members table CREATE TABLE members ( member_id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(255) NOT NULL, email VARCHAR(255) UNIQUE NOT NULL, joined DATE DEFAULT CURDATE() ); -- Borrow records (junction table) CREATE TABLE borrows ( borrow_id INT PRIMARY KEY AUTO_INCREMENT, member_id INT REFERENCES members(member_id), book_id INT REFERENCES books(book_id), borrow_date DATE DEFAULT CURDATE(), due_date DATE NOT NULL, returned BOOLEAN DEFAULT FALSE );
Read — SELECT with JOIN
-- All active borrows with member and book details SELECT b.borrow_id, m.name AS member_name, bk.title AS book_title, b.borrow_date, b.due_date, DATEDIFF(b.due_date, CURDATE()) AS days_remaining FROM borrows b JOIN members m ON b.member_id = m.member_id JOIN books bk ON b.book_id = bk.book_id WHERE b.returned = FALSE ORDER BY b.due_date ASC;
Update — mark as returned
UPDATE borrows SET returned = TRUE WHERE borrow_id = ?; UPDATE books SET status = 'available' WHERE book_id = ( SELECT book_id FROM borrows WHERE borrow_id = ? );
Aggregate — most borrowed books
SELECT bk.title, bk.author, COUNT(b.borrow_id) AS times_borrowed FROM books bk LEFT JOIN borrows b ON bk.book_id = b.book_id GROUP BY bk.book_id ORDER BY times_borrowed DESC LIMIT 5;
Constraint — overdue books
SELECT m.name, m.email, bk.title, DATEDIFF(CURDATE(), b.due_date) AS days_overdue FROM borrows b JOIN members m ON b.member_id = m.member_id JOIN books bk ON b.book_id = bk.book_id WHERE b.returned = FALSE AND b.due_date < CURDATE();
