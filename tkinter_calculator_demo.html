import tkinter as tk

# ── state ────────────────────────────
current  = ""
previous = ""
operator = None

def press(val):
    global current
    current += str(val)
    display.config(text=current)

def clear():
    global current, previous, operator
    current = previous = ""
    operator = None
    display.config(text="0")

def set_op(op):
    global current, previous, operator
    if current:
        previous = current
        current  = ""
        operator = op

def calculate():
    global current, previous, operator
    if not (previous and current and operator):
        return
    a, b = float(previous), float(current)
    result = {
        "+": a + b, "-": a - b,
        "*": a * b, "/": a / b if b else "Error"
    }.get(operator, "")
    r = int(result) if isinstance(result, float) \
        and result.is_integer() else result
    display.config(text=str(r))
    current = str(r); previous = ""

# ── window ───────────────────────────
root = tk.Tk()
root.title("Calculator")
root.resizable(False, False)

display = tk.Label(root, text="0",
    font=("Helvetica", 28),
    anchor="e", bg="#f5f5f5",
    width=12, relief="flat")
display.grid(row=0, column=0,
    columnspan=4, padx=10, pady=10)

# ── buttons ──────────────────────────
layout = [
    ["C","±","%","÷"],
    ["7","8","9","×"],
    ["4","5","6","−"],
    ["1","2","3","+"],
    ["0",".","="],
]

ops = {"÷":"/","×":"*","−":"-","+":"+"}

for r, row in enumerate(layout, 1):
    for c, lbl in enumerate(row):
        span = 2 if lbl == "0" else 1
        cmd = (clear if lbl=="C" else
               calculate if lbl=="=" else
               (lambda o=ops[lbl]: set_op(o))
                   if lbl in ops else
               (lambda v=lbl: press(v)))
        bg = ("#4a90d9" if lbl=="=" else
              "#e8f0fb" if lbl in ops else
              "#eee" if lbl in ("C","±","%") else "#fff")
        tk.Button(root, text=lbl, width=4,
            font=("Helvetica",16), bg=bg,
            command=cmd).grid(
            row=r, column=c,
            columnspan=span,
            padx=4, pady=4, ipady=8)

root.mainloop()
